float moveZ = Input.GetAxis("Vertical"); Vector3 move = transform.right * moveX + transform.forward * moveZ; Vector3 velocity = move * speed; velocity.y = rb.velocity.y; rb.velocity = velocity; if (Input.GetButtonDown("Jump")) rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse); } } public class PlayerCamera : MonoBehaviourPun { public Transform playerBody; public float mouseSensitivity = 100f; private float xRotation = 0f; void Start() { if (!photonView.IsMine) { Destroy(this); currentHealth -= amount; if (healthUI != null) healthUI.SetHealth(currentHealth / maxHealth); if (currentHealth <= 0f) Die(); } void Die() { currentHealth = maxHealth; transform.position = Vector3.zero; // respawn at origin if (healthUI != null) healthUI.SetHealth(currentHealth / maxHealth); } } #endregion #region WEAPONS MODULE public class Gun : MonoBehaviourPun { public float damage = 20f; public float range = 50f; if (target != null) target.TakeDamage(damage); } } } public class WeaponPickup : MonoBehaviour { public Gun weaponPrefab; void OnTriggerEnter(Collider other) { if (other.CompareTag("Player")) { Gun newGun = Instantiate(weaponPrefab); } public override void OnJoinedLobby() { PhotonNetwork.JoinOrCreateRoom("Room1", new RoomOptions { MaxPlayers = 10 }, TypedLobby.Default); } public override void OnJoinedRoom() { PhotonNetwork.Instantiate(playerPrefab.name, Vector3.zero, Quaternion.identity); } void Update() { if (!photonView.IsMine) { transform.position = Vector3.Lerp(transform.position, networkPos, Time.deltaTime * 10); transform.rotation = Quaternion.Lerp(transform.rotation, networkRot, Time.deltaTime * 10); } } public void OnPhotonSerializeView(PhotonStream stream, PhotonMessageInfo info) { if (stream.IsWriting) { stream.SendNext(transform.position); stream.SendNext(transform.rotation); } else { networkPos = (Vector3)stream.ReceiveNext();
